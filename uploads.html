// Khởi tạo Firebase const app = initializeApp(firebaseConfig); const storage = getStorage(app); const db = getFirestore(app); // 3. Hàm Upload File window.handleUpload = async () => { const fileInput = document.getElementById("fileInput"); const status = document.getElementById("status"); const btn = document.getElementById("btnUpload"); if (fileInput.files.length === 0) return alert("Choose file to upload!"); const file = fileInput.files[0]; btn.disabled = true; status.innerText = "Uploading ...please wait!"; try { // 1. Lưu file vào Storage const storageRef = ref(storage, 'gallery/' + Date.now() + "_" + file.name); await uploadBytes(storageRef, file); const url = await getDownloadURL(storageRef); // 2. Lưu link vào Firestore await addDoc(collection(db, "user_uploads"), { imageUrl: url, timestamp: new Date() }); status.innerText = "Upload successfully!"; fileInput.value = ""; // Reset ô chọn file } catch (e) { console.error(e); status.innerText = "ERROR: " + e.message; } finally { btn.disabled = false; } }; // Hàm hiển thị ảnh thời gian thực const q = query(collection(db, "user_uploads"), orderBy("timestamp", "desc")); onSnapshot(q, (snapshot) => { const gallery = document.getElementById("gallery"); gallery.innerHTML = ""; // Xóa cũ hiện mới snapshot.forEach((doc) => { const data = doc.data(); const img = document.createElement("img"); img.src = data.imageUrl; gallery.appendChild(img); }); });
